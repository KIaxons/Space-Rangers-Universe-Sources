    CustomArtefacts ^{
        TechDirector ^{
			Name=Тех-директор
			Broken=(расфокусирован)
            BrokenInBattle=(расфокусирован)
            BrokenInUse=(расфокусирован)
            Size=5
			Cost=2500
            Text=Позволяет перенастраивать доминаторский <color=255,240,100>тех-детектор</color> на частоту доминаторов <color=255,240,100>любой</color> серии.
			OnActCodeTypes=t_OnCheckingUsability,t_OnAnotherItem
            OnActCode ^{
				01=dword cur_item = CurItem();
				02=dword scaner = ScriptItemActObject1();
				//Если тех-директор не слишком изношен
				03=if(ItemDurability(cur_item) < 10)
				04 ^{
					01=if(ItemType(scaner) == t_Scaner)
					02 ^{
						01=if(ItemOwner(scaner) == Kling)
						02 ^{
							01=if(ScriptItemActionType(t_OnCheckingUsability))
							02 ^{
								01=ScriptItemActParam(1);
								02=exit;
							}
							03=else
							04 ^{
								01=unknown tarr = newarray(2);
								02=int sub_race = ItemSubrace(scaner);
								03=int serie_set = 0;
								04=if(sub_race != 0)
								05 ^{
									01=tarr[0] = '<color=220,0,0>Блазероидная</color>';
									02=serie_set = serie_set + 1;
								}
								06=if(sub_race != 1)
								07 ^{
									01=if(!serie_set) tarr[0] = '<color=0,140,220>Келлероидная</color>';
									02=else tarr[1] = '<color=0,140,220>Келлероидная</color>';
									03=serie_set = serie_set + 2;
								}
								08=if(sub_race != 2)
								09 ^{
									01=tarr[1] = '<color=50,140,15>Терроноидная</color>';
									02=serie_set = serie_set + 4;
								}
								10=int chosen_serie = ListBox('Выберите желаемую частоту:', tarr);
								11=if(chosen_serie > -1)
								12 ^{
									01=str serie_name;
									02=if(!chosen_serie)
									03 ^{
										01=if((serie_set & 1)) serie_name = 'блазероидную';
										02=else
										03 ^{
											01=serie_name = 'келлероидную';
											02=chosen_serie = 1;
										}
									}
									04=else
									05 ^{
										01=if((serie_set & 2)) serie_name = 'келлероидную';
										02=else
										03 ^{
											01=serie_name = 'терроноидную';
											02=chosen_serie = 2;
										}
									}
									06=ItemSubrace(scaner, chosen_serie);
									07=Sound('Sound.SaveLoadLoop');
									08=ItemDurability(cur_item, ItemDurability(cur_item) - Rnd(10, 20));
									09=if(ItemDurability(cur_item) < 0) ItemDurability(cur_item, 0);
									10=MessageBox(Format('<color=0,50,200>Тех-детектор</color> перенастроен на #0# частоту.', '#0#', serie_name));
								}
							}
						}
						03=else if(ScriptItemActionType(t_OnAnotherItem)) ScriptItemActParam(2);
					}
					03=else if(ScriptItemActionType(t_OnAnotherItem)) ScriptItemActParam(2);
				}
				05=MessageBox('<color=0,50,200>Тех-директор</color> расфокусирован, перенастройка сканера невозможна.');
            }
        }
    }
