					11 ^{
						01=if(ShipCustomFaction(s)=='Klissan')
						02 ^{
							01=if(ItemName(i)=='Ретрактор')dmg=150;
							02=else if(ItemName(i)=='Фотонная пушка М2')dmg=150;
